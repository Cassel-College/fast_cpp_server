19// =============================================================================
20// MyFlyControl — 飞控通信业务层
21//
22// 功能概述：
23// 1. 通过 JSON 配置初始化串口连接
24// 2. 启动后台接收线程，持续从串口读取数据并拆帧
25// 3. 提供各类飞行控制指令的发送接口
26// 4. 维护最新的飞控心跳状态，线程安全可读
27// 5. 支持注册回调：心跳更新、指令回复、云台控制
28// 6. 指令发送支持超时等待回复（2秒超时）
29//
30// 使用示例：
31// fly_control::MyFlyControl fc;
32// fc.Init(json_config);
33// fc.Start();
34// fc.SetOnHeartbeat([](const HeartbeatData& hb){ ... });
35// fc.SendSetSpeed(1500); // 速度 15.00 m/s
36// auto status = fc.GetLatestHeartbeat();
37// fc.Stop();
38// =============================================================================
39
